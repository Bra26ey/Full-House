USER_SCHEMA=CREATE TABLE IF NOT EXISTS user(id INT NOT NULL AUTO_INCREMENT, login VARCHAR(20) NOT NULL, password VARCHAR(50) NOT NULL, username VARCHAR(20) NOT NULL, avatar VARCHAR(255), money DOUBLE NOT NULL, PRIMARY KEY (id), UNIQUE (login));
BOARD_SCHEMA=CREATE TABLE IF NOT EXISTS board(id INT NOT NULL AUTO_INCREMENT, admin_id INT NOT NULL, password VARCHAR(50), button_pos INT DEFAULT(0), small_blind_pos INT DEFAULT(0), big_blind_pos INT DEFAULT(0), small_blind_bet INT DEFAULT(0), big_blind_bet INT DEFAULT(0), max_size_of_players INT DEFAULT(0), count_of_player_cards INT DEFAULT(0), PRIMARY KEY (id), INDEX (admin_id), FOREIGN KEY (admin_id) REFERENCES user (id) ON DELETE CASCADE);
ACTIVE_BOARD_SCHEMA=CREATE TABLE IF NOT EXISTS active_board(board_id INT NOT NULL, player_id INT NOT NULL, reserved_money DOUBLE, position INT DEFAULT(0), PRIMARY KEY (board_id, player_id), FOREIGN KEY (board_id) REFERENCES board (id) ON DELETE CASCADE, FOREIGN KEY (player_id) REFERENCES user (id) ON DELETE CASCADE);
